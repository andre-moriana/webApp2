    // Méthodes pour les exercices
    public function getExercises() {
        return $this->makeRequest("exercise-sheets");
    }
    
    public function getExerciseById($exerciseId) {
        return $this->makeRequest("exercise-sheets/" . $exerciseId);
    }
    
    public function createExercise($data) {
        return $this->makeRequest("exercise-sheets", "POST", $data);
    }
    
    public function updateExercise($exerciseId, $data) {
        return $this->makeRequest("exercise-sheets/" . $exerciseId, "PUT", $data);
    }
    
    public function deleteExercise($exerciseId) {
        return $this->makeRequest("exercise-sheets/" . $exerciseId, "DELETE");
    }
    
    public function getExerciseCategories() {
        return $this->makeRequest("exercise-sheets/categories");
    }
    
    public function createExerciseCategory($data) {
        return $this->makeRequest("exercise-sheets/categories", "POST", $data);
    }
    
    public function updateExerciseCategory($categoryId, $data) {
        return $this->makeRequest("exercise-sheets/categories/" . $categoryId, "PUT", $data);
    }
    
    public function deleteExerciseCategory($categoryId) {
        return $this->makeRequest("exercise-sheets/categories/" . $categoryId, "DELETE");
    }
